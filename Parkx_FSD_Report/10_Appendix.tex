\chapter{APPENDIX}

\section{Minimum Software Requirements}
\medskip
\textbf{Operating System}: Windows 10/11, macOS, or Linux
\\\\
\textbf{Web Server}: XAMPP (Apache) or Laravel Herd
\\\\
\textbf{Database Engine}: PostgreSQL 13+
\\\\
\textbf{Language/Frameworks}: PHP 8.2+, Laravel 12+, Node.js (v18+)

\section{Minimum Hardware Requirements}
\textbf{Processor}: Intel Core i3 (or AMD equivalent) or higher to handle concurrent local development processes.
\\\\
\textbf{RAM}: 4 GB (8 GB recommended for simultaneous IDE and server rendering)
\\\\
\textbf{Storage}: 1 GB free space for project files and \texttt{vendor}/\texttt{node\_modules} dependencies.
\\\\
\textbf{Display}: A standard HD monitor for clear layout debugging.
\\\\
\textbf{Input Devices}: Standard keyboard and mouse/trackpad.
\\\\
\textbf{Network Connectivity}: Required initially to fetch Composer dependencies, run NPM installs, and dynamically load external font weights.
